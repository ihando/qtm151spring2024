    if (Overall > 90):
        return "Top Performer"
    else:
        return "Non Top-Performer"

# Write your own code
plt.scatter(x = fifa["Potential"], y = fifa["Age"])
model = smf.ols(formula = 'Age ~  Potential', data = fifa)
results = model.fit()
b_list = results.params
fifa["p_estimated"] = b_list[0] + b_list[1]  * fifa["Potential"]
plt.plot(fifa["Potential"],fifa["p_estimated"], color = 'green')

plt.ylabel("Age")
